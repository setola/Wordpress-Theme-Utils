 54         /**
 55          * Prints the metabox markup
 56          * @param $post the current post
 57          * TODO: support for other input types and select :D
 58          */
 59         public function metabox_html($post) {
 60             wp_nonce_field($this->metaname, $this->metaname . '_nonce');
 61 
 62             $values = $this->get_meta($post->ID);
 63 
 64             $rows = '';
 65             foreach ($this->fields as $field) {
 66                 $th = HtmlHelper::standard_tag('th', $field['label'], array('class' => ''));
 67 
 68                 switch ($field['type']) {
 69                     case 'callback':
 70                         $input = $this->_markup_callback($field, $values, $post);
 71                         break;
 72 
 73                     case 'select':
 74                         $input = $this->_markup_select($field, $values);
 75                         break;
 76 
 77                     case 'single-attachment':
 78                         $input = $this->_markup_single_attachment($field, $values);
 79                         break;
 80 
 81                     case 'chebox-list':
 82                         $input = $this->_markup_checkbox_list($field, $values);
 83                         break;
 84 
 85                     case 'checkbox':
 86                         $input = $this->_markup_checkbox($field, $values);
 87                         break;
 88 
 89                     case 'text':
 90                     default:
 91                         $input = $this->_markup_text($field, $values);
 92                         break;
 93                 }
 94 
 95                 if (!empty($field['description'])) {
 96                     $input .= HtmlHelper::br() . HtmlHelper::paragraph(
 97                             $field['description'], array('class' => 'description')
 98                         );
 99                 }
100 
101                 $td = HtmlHelper::standard_tag('td', $input, array('class' => ''));
102                 $tr = HtmlHelper::standard_tag('tr', $th . "\n" . $td);
103                 $rows .= $tr;
104             }
105 
106             echo HtmlHelper::standard_tag(
107                 'table', HtmlHelper::standard_tag('tbody', $rows), array('class' => 'form-table')
108             );
109         }

111         public function _markup_callback($field, $values, $post){
112             if (is_callable($field['options']['callback'])) {
113                 $box = isset($field['options']['box']) ? $field['options']['box'] : array();
114                 $input = call_user_func_array(
115                     $field['options']['callback'],
116                     array($post, $box, $values)
117                 );
118             }
119             return $input;
120         }

158         public function _markup_checkbox_list($field, $values){
159             $input = '';
160             if (count($field['options'])) {
161                 foreach ($field['options'] as $key => $option) {
162                     $input .= HtmlHelper::input(
163                         $this->metaname . '[' . $field['id'] . '][' . $key . ']',
164                         'checkbox',
165                         array(
166                             'id'      => 'chebox-list-' . $field['id'] . '-' . $key,
167                             'class'   => '',
168                             'checked' => empty($values[$field['id']][$key]) ? '' : 'checked'
169                         )
170                     ) . PHP_EOL . HtmlHelper::label(
171                         $option, 'chebox-list-' . $field['id'] . '-' . $key
172                     ) . PHP_EOL . HtmlHelper::br();
173                 }
174             }
175             return $input;
176         }

215         /**
216          * Saves the metabox data while saving the page
217          */
218         public function save_metabox_data($post_id) {
219             if (!isset($post_id))
220                 return;
221             if ( defined('DOING_AUTOSAVE') && DOING_AUTOSAVE )
222                 return $post_id;
223             if (!isset($_POST['post_type']))
224                 return $post_id;
225             // First we need to check if the current user is authorised to do this action.
226             if ('page' == $_POST['post_type']) {
227                 if (!current_user_can('edit_page', $post_id))
228                     return;
229             }
230             else {
231                 if (!current_user_can('edit_post', $post_id))
232                     return;
233             }
234 
235             // Secondly we need to check if the user intended to change this value.
236             wp_verify_nonce($_POST[$this->metaname . '_nonce'], $this->metaname);
237             // TODO: understand how to make this check working :|
238             //check_admin_referer($this->metaname, $this->metaname . '_nonce');
239 
240             array_walk_recursive(
241                 $_POST[$this->metaname], array(
242                     __CLASS__,
243                     'sanitize_r'
244                 )
245             );
246 
247             update_post_meta($post_id, $this->metaname, $_POST[$this->metaname]);
248         }

254         /**
255          * Retrieves the post meta content.
256          * If a field is not set or empty, the value of the array
257          * element with such key will be an empty string,
258          * just to avoid warnings on non illegal offsets
259          * @param $post_id
260          * @return array unserialized values for the meta
261          */
262         public function get_meta($post_id = null) {
263             if (is_null($post_id))
264                 $post_id = get_the_ID();
265             $values = get_post_meta($post_id, $this->metaname, true);
266 
267             foreach ($this->fields as $field) {
268                 if (empty($values[$field['id']])) {
269                     $values[$field['id']] = '';
270                 }
271             }
272 
273             return $values;
274         }
