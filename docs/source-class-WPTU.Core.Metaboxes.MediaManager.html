  8 /**
  9  * Manages multiple media set
 10  *
 11  * This feature adds a metabox to your WordPress admin panel
 12  * in which you can add any media media into a given set.
 13  *
 14  * Then in your theme you will be able to retrieve the single
 15  * set and show it where you need it.
 16  *
 17  * For example if you need to have a slideshow on the top of the page
 18  * and a minigallery on the bottom, this feature is what you need
 19  * to separate your images between the two sets.
 20  *
 21  * In your functions.php simply add
 22  * <code>
 23  *  MediaManager::enable();
 24  *  MediaManager::set_media_list('slideshow', array(
 25  *      'label'     =>  __('Slideshow', 'wtu_framework'),
 26  *      'shortcode' =>  'mySlideshow'
 27  *  ));
 28  * </code>
 29  * to enable the backend feature.
 30  *
 31  * Now you will have to manage a new shortcode; for example:
 32  *
 33  * <code>
 34  *  function mySlideshow_gallery_shortcode($atts){
 35  *      $data = shortcode_atts(array('ids' => ''), $atts );
 36  *      $toret = '';
 37  *      $ids = explode(',', $data['ids']);
 38  *      foreach($ids as $id){
 39  *          $toret .= wp_get_attachment_image($id);
 40  *      }
 41  *      return $toret;
 42  *  }
 43  *
 44  *  add_shortcode('mySlideshow', 'mySlideshow_gallery_shortcode');
 45  * </code>
 46  *
 47  * If you need more than a section simply duplicate your code and change
 48  * at least the shortcode and the id (first parameter of set_media_list())
 49  *
 50  * @author Emanuele 'Tex' Tessore
 51  * @version 1.0.0
 52  *
 53  */

136

205      * Registers the metabox
206      *
207      * @uses add_meta_box
208      *
209      * @param string $post_type The type of Write screen on
210      * which to show the edit screen section ('post', 'page', 'dashboard',
211      * 'link', 'attachment' or 'custom_post_type' where custom_post_type
212      * is the custom post type slug)
213      *
214      * @param string $context The part of the page where the edit screen
215      * section should be shown ('normal', 'advanced', or 'side').
216      * (Note that 'side' doesn't exist before 2.7)
217      *
218      * @param string $priority The priority within the context where the
219      * boxes should show ('high', 'core', 'default' or 'low')
220      */
221     public static function register_metaboxes($post_type='', $context='side', $priority='high'){
222         global $post;
223         $template = self::get_template($post);
224         $show_meta_box = false;
225 
226         foreach(self::$media_list as $k => $elem){
227             $template_checker = new TemplateChecker();
228             if(isset($elem['include'])) $template_checker->set_include($elem['include']);
229             if(isset($elem['exclude'])) $template_checker->set_exclude($elem['exclude']);
230             if($template_checker->check($template)) {
231                 $show_meta_box = true;
232                 break;
233             }
234         }
235 
236         if($show_meta_box){
237             add_meta_box(
238                 'wpu-media-manager',
239                 __( 'Galleries', 'wtu_framework' ),
240                 array(__CLASS__, 'metabox_html'),
241                 $post_type,
242                 $context,
243                 $priority
244             );
245         }
246     }

390     /**
391      * Saves the metabox data while saving the page
392      * @param int $post_id the post id
393      * @deprecated using ajax instead
394      */
395     public static function save_metabox_data($post_id){
396         if(!isset($post_id)) return;
397         if(!isset($_POST['post_type'])) return;
398         // First we need to check if the current user is authorised to do this action.
399         if ( 'page' == $_POST['post_type'] ) {
400             if ( ! current_user_can( 'edit_page', $post_id ) )
401                 return;
402         } else {
403             if ( ! current_user_can( 'edit_post', $post_id ) )
404                 return;
405         }
406 
407         // Secondly we need to check if the user intended to change this value.
408         if (
409             !isset($_POST[self::META_KEY_NAME.'_nonce'])
410             || !wp_verify_nonce($_POST[self::META_KEY_NAME.'_nonce'], __FILE__)
411         ) {
412             return;
413         }
414 
415         $sanitized = array();
416         foreach($_POST[self::META_KEY_NAME] as $k => $v){
417             if(is_array($v)){
418                 foreach($v as $k1 => $v1){
419                     $v[sanitize_text_field($k1)] = sanitize_text_field(trim($v1, '"'));
420                 }
421                 $sanitized[sanitize_text_field($k)] = $v;
422             } else {
423                 $sanitized[sanitize_text_field($k)] = sanitize_text_field(trim($v, '"'));
424             }
425         }
426 
427         update_post_meta(
428             $_POST['post_ID'],
429             self::META_KEY_NAME,
430             $sanitized
431         );
432     }

472     /**
473      * Utitlity to retrieve the post meta for the given media set.
474      *
475      * If WPML is available and notthing is retrieved from the
476      * given $post_id, this method will search for the same
477      * post meta in the corresponding post in default language.
478      *
479      * It is protected cause it can be overloaded by child classes
480      * but it's for internal use only. MediaManager::get_media()
481      * is the public method.
482      *
483      * @uses icl_object_id() (WPML) if available
484      * @param string $set identifier of the set of elements
485      * @param int $post_id the post ID to be queried
486      * @return string the meta
487      */
488     protected static function get_meta($set, $post_id){
489         $meta_name = self::META_KEY_NAME.'-'.sanitize_title($set);
490         $data = get_post_meta($post_id, $meta_name, true);
491 
492         if(isset($GLOBALS['sitepress'])){
493             global $sitepress;
494 
495             // look for the post in default language
496             if(self::$media_list[$set]['wpml']['default_lang'] && empty($data)){
497                 $post_id = icl_object_id($post_id, get_post_type($post_id), true, $sitepress->get_default_language());
498                 if($post_id == 0) return;
499                 $data = get_post_meta($post_id, $meta_name, true);
500             }
501 
502             // look for the homepage in current language
503             if(self::$media_list[$set]['wpml']['default_lang'] && empty($data)){
504                 $data = get_post_meta(get_option('page_on_front'), $meta_name, true);
505             }
506 
507             // look for the homepage in default language
508             if(self::$media_list[$set]['wpml']['default_lang'] && empty($data)){
509                 $post_id = icl_object_id(get_option('page_on_front'), get_post_type(get_option('page_on_front')), true, $sitepress->get_default_language());
510                 if($post_id == 0) return;
511                 $data = get_post_meta($post_id, $meta_name, true);
512             }
513         }
514 
515         return $data;
516     }
