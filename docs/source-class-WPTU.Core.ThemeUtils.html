 54     /**
 55      * Enables the given feature
 56      * @param $feature_name int one of the class FEATURE_* constants
 57      * @return $this ThemeUtils for chainability
 58      * @throws \Exception if the given feature is already enabled
 59      */
 60     public function enable_feature($feature_name) {
 61 
 62         if (in_array($feature_name, $this->enabled_features))
 63             throw new \Exception(sprintf('Feature "%s" is already enabled', $feature_name));
 64 
 65         if(is_array($feature_name)){
 66             foreach($feature_name as $fn){
 67                 $this->_enable_feature($fn);
 68             }
 69         } else {
 70             $this->_enable_feature($feature_name);
 71         }
 72 
 73         return $this;
 74     }
 75 
 76     /**
 77      * @param $feature_name
 78      * @throws \Exception if $featureName is not recognized
 79      */
 80     protected function _enable_feature($feature_name) {
 81         switch ($feature_name) {
 82             case self::FEATURE_AUTOLOAD:
 83                 $this->enable_autoload_system();
 84                 break;
 85             case self::FEATURE_MEDIAMANAGER:
 86                 $this->enable_media_manager();
 87                 break;
 88             case self::FEATURE_DEBUG:
 89                 $this->enable_debug();
 90                 break;
 91             case self::FEATURE_DUMMYCONTENT:
 92                 $this->enable_dummy_content();
 93                 break;
 94             default:
 95                 throw new \Exception('Feature not found');
 96                 break;
 97         }
 98     }

131     protected function enable_autoload_system() {
132         $this->enabled_features[] = self::FEATURE_AUTOLOAD;
133         ClassAutoloader::getInstance()
134             // First search in the child theme dir
135             ->addLoadingTemplate(get_stylesheet_directory() . '/%classname%.class.php')
136             // then search into the parent theme dir
137             ->addLoadingTemplate(get_template_directory() . '/%classname%.class.php')
138             // temporary only: check the todo-move folder for old code
139             ->addLoadingTemplate(get_template_directory() . '/WPTU/todo-move/%classname%.class.php');
140 
141         return $this;
142     }

178     /**
179      * Register some constants
180      * @return $this ThemeUtils for chainability
181      */
182     protected function default_constants() {
183         // initialize constants only once
184         if (defined('WORDPRESS_THEME_UTILS_PATH')) return $this;
185 
186         /**
187          * The absolute base path for Wordpress Theme Utils
188          */
189         if (!defined('WORDPRESS_THEME_UTILS_PATH'))
190             define('WORDPRESS_THEME_UTILS_PATH', dirname(dirname(__FILE__)) . DIRECTORY_SEPARATOR);
191 
192         /**
193          * The main directory name for Wordpress Theme Utils
194          */
195         if (!defined('WORDPRESS_THEME_UTILS_DIRNAME'))
196             define('WORDPRESS_THEME_UTILS_DIRNAME', basename(WORDPRESS_THEME_UTILS_PATH));
197 
198         /**
199          * Relative path for template parts
200          */
201         if (!defined('WORDPRESS_THEME_UTILS_PARTIALS_RELATIVE_PATH'))
202             define('WORDPRESS_THEME_UTILS_PARTIALS_RELATIVE_PATH', 'partials' . DIRECTORY_SEPARATOR);
203 
204         /**
205          * Path for libraries
206          */
207         if (!defined('WORDPRESS_THEME_UTILS_LIBRARIES_RELATIVE_PATH'))
208             define('WORDPRESS_THEME_UTILS_LIBRARIES_RELATIVE_PATH', 'libraries' . DIRECTORY_SEPARATOR);
209 
210         if (!defined('WORDPRESS_THEME_UTILS_LIBRARIES_ABSOLUTE_PATH'))
211             define(
212             'WORDPRESS_THEME_UTILS_LIBRARIES_ABSOLUTE_PATH',
213                 WORDPRESS_THEME_UTILS_PATH . DIRECTORY_SEPARATOR . 'libraries' . DIRECTORY_SEPARATOR
214             );
215 
216         /**
217          * Relative path for autoloader class
218          */
219         if (!defined('WORDPRESS_THEME_UTILS_AUTOLOADER_RELATIVE_PATH'))
220             define(
221             'WORDPRESS_THEME_UTILS_AUTOLOADER_RELATIVE_PATH',
222                 DIRECTORY_SEPARATOR . WORDPRESS_THEME_UTILS_DIRNAME . DIRECTORY_SEPARATOR . 'ClassAutoloader.class.php'
223             );
224 
225         if (!defined('WORDPRESS_THEME_UTILS_TEXTDOMAIN'))
226             define('WORDPRESS_THEME_UTILS_TEXTDOMAIN', 'wtu_framework');
227 
228         return $this;
229     }
