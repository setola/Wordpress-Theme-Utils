  7 /**
  8  * Merge all the images into a big one and serve it as div background
  9  * 
 10  * Takes a list of n images with the same dimensions
 11  * and merges them in a unique image with sizes (n x width) and height.
 12  * Than renders an html dom to serve the big image as background of some divs
 13  * so that you can cycle it.
 14  * 
 15  * @author etessore
 16  * @version 1.0.0
 17  *
 18  */

 20 class MergedSlideshow extends OneImageForAll{
 21     function __construct(){
 22         parent::__construct();
 23 
 24         $tpl = (is_front_page() || has_post_thumbnail())
 25         ? <<< EOF
 26             <div class="images_list">%images_list%</div>
 27             %loading%
 28 EOF
 29 : <<< EOF
 30             <div class="images_list">%images_list%</div>
 31             %loading%
 32             %next%
 33             %prev%
 34 EOF;
 35         //$this->force_image_refresh = true;
 36         $this
 37             ->set_wp_media_dimension($this->get_slideshow_dimension())
 38             ->get_the_images()
 39             ->set_uid('slideshow')
 40             ->set_markup(
 41                     'loading',
 42                     '<div class="loading" data-description="'
 43                     .__('Loading image %number% of %total%', 'theme')
 44                     .'"></div>'
 45             )
 46             ->set_markup('next', '<div id="go-next" class="sprite next"></div>')
 47             ->set_markup('prev', '<div id="go-prev" class="sprite prev"></div>')
 48             ->set_template($tpl);
 49     }
 50 
 51     /**
 52      * @return string the media gallery size for the slideshow
 53      */
 54     function get_slideshow_dimension(){
 55         global $post;
 56         $toret = 'slideshow';
 57         if(get_post_meta( $post->ID, '_wp_page_template', true ) == 'offers.php')
 58             $toret .= '-small';
 59         if(is_front_page())
 60             $toret .= '-home';
 61 
 62         return $toret;
 63     }

 66     /**
 67      * Retrives the images for the slideshow
 68      * @param int $post_id the post id to dig in
 69      * @return MergedSlideshow $this for chainability
 70      */
 71     function get_the_images($post_id = null){
 72 
 73         $this->post_id = ($post_id) ? $post_id : get_the_ID();
 74         $this->images = get_posts(
 75                 array(
 76                         'post_parent'   => $this->post_id,
 77                         'post_type'     => 'attachment',
 78                         'post_mime_type'=> 'image',
 79                         'orderby'       => 'menu_order',
 80                         'order'         => 'ASC',
 81                         'tax_query'     =>  array(
 82                                 'taxonomy'      => 'media_tag',
 83                                 'field'         => 'slug',
 84                                 'terms'         => 'slideshow',
 85                                 'operator'      => 'IN'
 86                         )
 87                 )
 88         );
 89 
 90         if(empty($this->images)){
 91             $this->images = get_posts(
 92                     array(
 93                             'post_parent'   => get_option('page_on_front'),
 94                             'post_type'     => 'attachment',
 95                             'tax_query'     =>  array(
 96                                     'taxonomy'      => 'media_tag',
 97                                     'field'         => 'slug',
 98                                     'terms'         => 'slideshow',
 99                                     'operator'      => 'IN'
100                             )
101                     )
102             );
103         }
104 
105         return $this;
106     }
