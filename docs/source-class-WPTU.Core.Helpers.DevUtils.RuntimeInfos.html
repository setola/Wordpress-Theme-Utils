 15  * Changelog

 18  */
 19 class RuntimeInfos{
 20     /**
 21      * @var array the list of infos
 22      */
 23     public $infos = array();
 24     
 25     /**
 26      * @var int the way we want the infos
 27      */
 28     public $type;
 29     
 30     /**
 31      * @var int id of the post these infos refer to
 32      */
 33     protected $id;
 34     
 35     /**
 36      * 
 37      * @var int use this as type to have a <script> tag in the footer
 38      */
 39     const TYPE_FOOTER = 1;
 40     
 41     /**
 42      * @var int uset this as type to have an external js on an admin-ajax.php url
 43      */
 44     const TYPE_AJAX = 2;
 45     
 46     /**
 47      * @var string the action parameter for the admin-ajax
 48      */
 49     const ajax_action = 'runtime-infos';
 50     
 51     /**
 52      * Initializes the current set of infos with default values
 53      */
 54     public function __construct(){
 55         $this->set_type(self::TYPE_AJAX);
 56     }
 57     
 58     /**
 59      * Sets the default values: 
 60      * postID, theme_url, upload_dir, home_url and ajaxurl
 61      */
 62     public function set_standard_infos(){
 63         $uploadDir = wp_upload_dir();
 64         $uploadDir = $uploadDir['baseurl'];
 65         $this
 66             ->set_type(self::TYPE_AJAX)
 67             ->add_info('postID', $this->get_the_ID())
 68             ->add_info('theme_url', get_template_directory_uri())
 69             ->add_info('upload_dir', $uploadDir)
 70             ->add_info('home_url', get_bloginfo('url'))
 71             ->add_info('ajaxurl', admin_url('admin-ajax.php'));
 72         
 73         if(defined('ICL_LANGUAGE_CODE')){
 74             $this->add_info('currentLanguage', substr(ICL_LANGUAGE_CODE, 0, 2));
 75         }
 76         
 77         $this->generate_unique_id();
 78 
 79         set_transient($this->id, $this->infos);
 80         if($this->type == self::TYPE_AJAX){
 81             $this->load_assets();
 82         }
 83     }
 84     
 85     /**
 86      * Get the ID for the current post
 87      */
 88     private function get_the_ID(){
 89         if(is_404()) return '';
 90         return get_the_ID();
 91     }
 92     
 93     /**
 94      * Adds an info to the current set
 95      * @param string $key the name of the info
 96      * @param string $value the value of the info
 97      * @return RuntimeInfos $this for chainability
 98      */
 99     public function add_info($key, $value){
100         $this->infos[$key] = $value;
101         return $this;
102     }
103     
104     /**
105      * Sets the type
106      * @param int $type
107      * @see RuntimeInfos::TYPE_AJAX
108      * @see RuntimeInfos::TYPE_FOOTER
109      */
110     public function set_type($type){
111         $this->type = $type;
112         return $this;
113     }
114     
115     /**
116      * Hook the print of this infos to Wordpress footer
117      * @return RuntimeInfos $this for chainability
118      */
119     public function hook(){
120         switch($this->type){
121             case self::TYPE_AJAX:
122                 add_action('wp_ajax_'.self::ajax_action, array(__CLASS__, 'ajax_callback'));
123                 add_action('wp_ajax_nopriv_'.self::ajax_action, array(__CLASS__, 'ajax_callback'));
124                 add_action('wp', array(&$this, 'set_standard_infos'));
125             break;
126             
127             case self::TYPE_FOOTER:
128                 add_action('wp_footer', array(&$this, 'the_markup'));
129             break;
130         }
131         return $this;
132     }
133     
134     /**
135      * Registers and enqueue the js from admin-ajax.php
136      * This has to be called after RuntimeInfos::generate_unique_id()
137      */
138     public function load_assets(){
139         if(!isset($this->id)){
140             wp_die('You have to call RuntimeInfos::generate_unique_id() before RuntimeInfos::load_assets()');
141         }
142         $nonce = wp_create_nonce("runtime_infos_nonce");
143         $src = 
144             admin_url('admin-ajax.php')
145             .'?'
146             .http_build_query(
147                 array(
148                     'action'    =>  self::ajax_action,
149                     'id'        =>  $this->id,
150                     'nonce'     =>  $nonce
151                 )
152             );
153         wp_register_script('ajax-runtime-infos', $src, false, false, false);
154         wp_enqueue_script('ajax-runtime-infos');
155     }
156     
157     /**
158      * Format the data set in JSON.
159      * Overload this if you want a different
160      * data structure, i.e. XML
161      */
162     protected function render(){
163         return json_encode($this->infos);
164     }
165     
166     /**
167      * Retrieves the html markup
168      * @return string html markup
169      */
170     function get_markup() {
171         return ThemeHelpers::script($this->render(), 'id="runtime-infos"');
172     }
173     
174     /**
175      * Generates an unique ID for the current data set
176      * @return string the ID
177      */
178     private function generate_unique_id(){
179         $this->id = abs(crc32(serialize($this->infos)));
180         return $this->id;
181     }
182     
183     /**
184      * Prints the markup
185      */
186     public function the_markup(){
187         echo $this->get_markup();
188     }
189     
190     /**
191      * Callback executed when the admin ajax url 
192      * is called with self::ajax_action parameter
193      */
194     public static function ajax_callback(){
195         $data = get_transient(intval($_GET['id']));
196         header("Content-type: text/javascript");
197         die('var runtimeInfos='.json_encode($data).';');
198     }
199 }
