 68     public function locate_favicon($names=array('favicon'), $extensions=array('ico','png'), $paths=array()){
 69         if(count($paths)==0){
 70             $images = '/images/';
 71             $paths[] = array(
 72                 'path'=>get_template_directory().$images, 
 73                 'url'=>get_template_directory_uri().$images
 74             );
 75             $paths[] = array(
 76                 'path'=>get_stylesheet_directory().$images, 
 77                 'url'=>get_stylesheet_directory_uri().$images
 78             );
 79         }
 80         
 81         foreach($paths as $path){
 82             foreach($extensions as $extension){
 83                 foreach($names as $name){
 84                     $fname = $name.'.'.$extension;
 85                     $file = $path['path'].$fname;
 86                     if(file_exists($file)) return $path['url'].$fname;
 87                 }
 88             }
 89         }
 90     }
