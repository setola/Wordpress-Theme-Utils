 25      * Useful if you vant to customize jQuery.cycle 2 behavior
 26      *

 36     /**
 37      * Renders a single element of the slideshow
 38      * @param int $k the index of the element in $this->images array
 39      * @return string html markup for the given element
 40      */
 41     private function render_element($k){
 42         $toret = HtmlHelper::image(
 43             $this->get_image_src($k), 
 44             array(
 45                 'alt'           =>  $this->get_image_alt($k),
 46                 'data-desc'     =>  $this->get_image_description($k),
 47                 'data-caption'  =>  $this->get_image_caption($k),
 48                 'id'            =>  $this->get_image_id($k),
 49                 'width'         =>  $this->get_image_width($k),
 50                 'height'        =>  $this->get_image_height($k)
 51             )
 52         );
 53         
 54         return $toret;
 55     }

 93         
 94         $script = HtmlHelper::script(json_encode($list), array('id'=>'script_'.$this->unid));
 95         
 96         $defaults = array(
 97             'id'                        =>  $this->unid, 
 98             'class'                     =>  'cycle-slideshow',
