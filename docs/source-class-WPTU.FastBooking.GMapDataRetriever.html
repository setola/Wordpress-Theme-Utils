 25      * @param array $map_data the map data

 33     /**
 34      * Set the class to get map datas from the Simple fields plugin
 35      * Checks if the plugin is enabled.
 36      * @return GMapDataRetriever $this for chainability
 37      */
 38     public function use_simple_fields(){
 39         if(!function_exists('simple_fields_get_post_group_values')){
 40             wp_die('You need Simple Fields to be Up And Running!');
 41         }
 42         
 43         $default = array(
 44                 'Map'           =>  array(
 45                     array(
 46                         'lat'   =>  0, 
 47                         'lng'   =>  0       
 48                     )
 49                 ),
 50                 'Title'         =>  array(''),
 51                 'Description'   =>  array(''),
 52                 'Balloon Text'  =>  array('')
 53         );
 54         
 55         $map_data = simple_fields_get_post_group_values(get_the_ID(),'Map Data');
 56         
 57         if(!empty($map_data)) ThemeHelpers::load_js('map');
 58         
 59         $map_data = wp_parse_args($map_data, $default);
 60         
 61         if(function_exists('simple_fields_field_googlemaps_register')){
 62             //vd($map_data);
 63             $data = array(
 64                 'center'    =>  array(
 65                         'lat'       =>  floatval($map_data['Map'][0]['lat']),
 66                         'lng'       =>  floatval($map_data['Map'][0]['lng'])
 67                 ),
 68                 'point'     =>  array(
 69                         'lat'       =>  floatval($map_data['Map'][0]['lat']),
 70                         'lng'       =>  floatval($map_data['Map'][0]['lng'])
 71                 ),
 72                 'zoom'      =>  intval($map_data['Map'][0]['preferred_zoom']),
 73                 'type'      =>  $map_data['Map Type'][0],
 74                 'title'     =>  $map_data['Title'][0],
 75                 'content'   =>  str_replace(
 76                         array('%book%'),
 77                         array('<a class="book-action" href="javascript:;">'.__('book','theme').'</a>'),
 78                         $map_data['Balloon Text'][0]
 79                 ),
 80                 'book_trans'    =>  __('book','theme')
 81             );
 82         } else {
 83             $data = array(
 84                 'center'    =>  array(
 85                     'lat'       =>  floatval(
 86                         empty($map_data['Center Latitude'][0])
 87                         ? $map_data['Latitude'][0]
 88                         : $map_data['Center Latitude'][0]
 89                     ),
 90                     'lng'       =>  floatval(
 91                         empty($map_data['Center Longitude'][0])
 92                         ? $map_data['Longitude'][0]
 93                         : $map_data['Center Longitude'][0]
 94                     )
 95                 ),
 96                 'point'     =>  array(
 97                     'lat'       =>  floatval($map_data['Latitude'][0]),
 98                     'lng'       =>  floatval($map_data['Longitude'][0])
 99                 ),
100                 'zoom'      =>  intval($map_data['Zoom'][0]),
101                 'type'      =>  $map_data['Map Type'][0],
102                 'title'     =>  $map_data['Balloon Title'][0],
103                 'content'   =>  str_replace(
104                     array('%book%'),
105                     array('<a class="book-action" href="javascript:;">'.__('book','theme').'</a>'),
106                     $map_data['Balloon Text'][0]
107                 ),
108                 'book_trans'    =>  __('book','theme')
109             );
110         }
111         
112         return $this->set_map_data($data);
113     }
