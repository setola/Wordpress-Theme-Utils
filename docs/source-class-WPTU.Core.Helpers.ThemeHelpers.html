 42     /**
 43      * This filter callback adds some usefull classes to the body
 44      * Remember to call body_class() in the theme!
 45      * 
 46      * @param array $classes the classes already added by wordpress or some other plugin
 47      */
 48     public static function body_class($classes){
 49         if(function_exists('fb_get_systags')){
 50             $tags = fb_get_systags(get_the_ID(), false);
 51         } else {
 52             $tags = array();
 53         }
 54         foreach($tags as $k => $tag){
 55             $tags[$k] = 'tag-'.$tag;
 56         }
 57         $classes = array_merge($classes, $tags);
 58         if(is_super_admin()) $classes[] = 'super-admin';
 59         if(!is_user_logged_in()) $classes[] = 'logged-out';
 60         
 61         return $classes;
 62     }

132     /**
133      * Merges some images to a single big one to save some http connections.
134      * Stores it in a cache folder.
135      * To disable the cache system set IMAGE_MERGE_FORCE_REFRESH constant to true
136      * @param array $images an array of images: for every element $image['path'] and $image['url'] have to be defined
137      * @param array $config timthumb config, default 'w'=>'700', 'h'=>'370', 'q'=>'50', 'r'=>false
138      * @return string the url for the big image
139      */
140     public static function merge_images($images, $config=null){
141         if(empty($images)) return 'No images';
142         $config = array_merge(
143             array(
144                 'w'     =>  '700',
145                 'h'     =>  '370',
146                 'q'     =>  '50',
147                 'r'     =>  false
148             ),
149             $config
150         );
151     
152         $combined_image = imagecreatetruecolor($config['w']*count($images), $config['h']);
153     
154         $cache_name = '';
155         foreach($images as $image){
156             $cache_name .= $image['path'].';';
157         }
158         $cache_name .= serialize($config);
159         $cache_name = md5($cache_name);
160     
161         $cache_dir = get_template_directory().'/cache/';
162         if (!@is_dir($cache_dir)){
163             if (!@mkdir($cache_dir)){
164                 die('Couldn\'t create cache dir: '.$cache_dir);
165             }
166         }
167         $cache_url = get_bloginfo('template_url').'/cache/'.$cache_name.'.jpg';
168         $cache_path = $cache_dir.$cache_name.'.jpg';
169     
170         if(
171             !file_exists($cache_path)
172             || IMAGE_MERGE_FORCE_REFRESH===true
173             || (
174                     get_option('development_mode','no') == 'yes'
175                     && $_GET['forcerefresh'] == 'true'
176             )
177         ){
178             foreach($images as $array_index => $image){
179                 $src = $image['url'].'?'.http_build_query($config, '', '&');
180     
181                 $info = getimagesize($src);
182                 switch($info['mime']){
183                     case 'image/jpeg':
184                         $image = imagecreatefromjpeg($src);
185                         break;
186                     case 'image/png':
187                         $image = imagecreatefrompng($src);
188                         break;
189                     case 'image/gif':
190                         $image = imagecreatefromgif($src);
191                         break;
192                     default:
193                         die('unknow mime type');
194                 }
195     
196                 imagecopymerge(
197                     $combined_image,
198                     $image,
199                     $array_index*$config['w'],
200                     0, 0, 0,
201                     $config['w'],
202                     $config['h'],
203                     100
204                 );
205     
206                 imagejpeg(
207                     $combined_image,
208                     $cache_path,
209                     $config['q']
210                 );
211             }
212         }
213     
214         return $cache_url;
215     }

217     /**
218      * Returns the content after the more tag for the given post
219      * @param string $more_link_text Optional. Content for when there is more text
220      * @param int|object $post_id Post ID or post object.
221      * @param bool $apply_filters true if you want the_content filter applyed on the return value
222      * @param bool $remove_more if true the function will remove the #more span added by WordPress
223      */
224     static function get_the_content_after_more($more_link_text = '', $post_id = null, $apply_filters = true, $remove_more = true){
225         $toret = '';
226         
227         if(isset($post_id)){
228             global $post;
229             $post = get_post($post_id);
230             setup_postdata($post);
231             $toret = get_the_content($more_link_text, true);
232             wp_reset_postdata();
233         } else {
234             $toret = get_the_content($more_link_text, true);
235         }
236         
237         if($apply_filters){
238             $toret = apply_filters('the_content', $toret);
239             $toret = str_replace(']]>', ']]&gt;', $toret);
240         }
241         
242         if($remove_more){
243             $toret = preg_replace('/<p>.*<span id="more-.*"><\/span>.*<\/p>/i', '', $toret);
244         }
245         
246         return $toret;
247     }

249     /**
250      * Prints the content before the more tag for the given post
251      * If the tag is not present it will print the entire body
252      * @param string $more_link_text Optional. Content for when there is more text
253      * @param int|object $post_id Post ID or post object.
254      * @param bool $apply_filters true if you want the_content filter applyed on the return value
255      */
256     static function get_the_content_before_more($more_link_text = '', $post_id = null, $apply_filters = true){
257         global $more, $post;
258         
259         $swap = $more;
260         $more = 0;
261         
262         if(isset($post_id)){
263             $post = get_post($post_id);
264             setup_postdata($post);
265             
266             $toret = get_the_content($more_link_text, false);
267             wp_reset_postdata();
268         } else {
269             $toret = get_the_content($more_link_text, false);
270         }
271         
272         $more = $swap;
273         
274         if($apply_filters){
275             $toret = apply_filters('the_content', $toret);
276             $toret = str_replace(']]>', ']]&gt;', $toret);
277         }
278         
279         return $toret;
280     }

371     /**
372      * Generates HTML Node Attribures
373      * @param string $glue
374      * @param array|string $pieces
375      * @return string
376      * @author http://blog.teknober.com/2011/04/13/php-array-to-html-attributes/
377      */
378     public static function array_to_html_attributes($glue, $pieces) {
379         _deprecated_function(__CLASS__.'::'.__FUNCTION__, '1.0.4');
380         $str = $pieces;
381         if (is_array($pieces)) {
382             $str = " ";
383             foreach($pieces as $key => $value) {
384                 if (strlen($value) > 0) {
385                     $str .= esc_attr($key) . esc_attr($glue) . '"' . esc_attr($value) . '" ';
386                 }
387             }
388         }
389     
390         return rtrim($str);
391     }

405     /**
406      * Retrieves the url from array $url_arr
407      * If defined use http_build_url, else custom code
408      * @param array $url_arr the array of attributes 
409      * @see http://php.net/manual/en/function.parse-url.php
410      */
411     public static function build_url($url_arr){
412         if(function_exists('http_build_url')){
413             return  http_build_url($url_arr);
414         } else {
415             $scheme   = isset($url_arr['scheme']) ? $url_arr['scheme'] . '://' : '';
416             $host     = isset($url_arr['host']) ? $url_arr['host'] : '';
417             $port     = isset($url_arr['port']) ? ':' . $url_arr['port'] : '';
418             $user     = isset($url_arr['user']) ? $url_arr['user'] : '';
419             $pass     = isset($url_arr['pass']) ? ':' . $url_arr['pass']  : '';
420             $pass     = ($user || $pass) ? "$pass@" : '';
421             $path     = isset($url_arr['path']) ? $url_arr['path'] : '';
422             $query    = isset($url_arr['query']) ? '?' . http_build_query($params) : '';
423             $fragment = isset($url_arr['fragment']) ? '#' . $url_arr['fragment'] : '';
424             return  "$scheme$user$pass$host$port$path$query$fragment";
425         }
426     }
