  9 /**
 10  * Generic abstract metaboxes controller: decides which metaboxes are to be shown on the various page of the website
 11  */
 12 abstract class MetaboxesController {
 13     /**
 14      * @var array stores the list of MetaboxInfo objects
 15      */
 16     protected $metabox_infos;
 17 
 18     /**
 19      * @var array list of metaboxes to be shown on the current page
 20      */
 21     protected $enabled_metaboxes = null;
 22 
 23     /**
 24      * Build a new object
 25      *
 26      * @param $metabox_infos array list of MetaboxInfo
 27      */
 28     protected function __construct( $metabox_infos ) {
 29         $this->metabox_infos = $metabox_infos;
 30         add_action( 'add_meta_boxes', array( &$this, 'register_admin' ) );
 31         add_action( 'save_post', array( &$this, 'save_metabox_data' ), 1 );
 32     }
 33 
 34     /**
 35      * Populates the $this->enabled_metaboxes list
 36      */
 37     public function filter_metaboxes() {
 38         if ( ! is_null( $this->enabled_metaboxes ) ) {
 39             return;
 40         }
 41 
 42         $this->enabled_metaboxes = array();
 43         $template                = get_post_meta( $_GET['post'], '_wp_page_template', true );
 44 
 45         foreach ( $this->metabox_infos as $metabox_info ) {
 46             $template_checker = new TemplateChecker( $metabox_info->positions['include'], $metabox_info->positions['exclude'] );
 47             if (
 48                 ( $template == 'default' || $template == '' )
 49                 && (
 50                     get_the_ID() == get_option( 'page_on_front' )
 51                     || get_option( 'page_on_front' ) == $_GET['post']
 52                     || get_option( 'page_on_front' ) == $_POST['post_ID']
 53                 )
 54             ) {
 55                 $template = 'front-page.php';
 56             }
 57 
 58             if ( $template_checker->check( $template ) ) {
 59                 $this->enabled_metaboxes[] = $metabox_info->metabox;
 60             }
 61         }
 62     }
 63 
 64     /**
 65      * Register the metaboxes to the WordPress admin page
 66      */
 67     public function register_admin() {
 68         $this->filter_metaboxes();
 69         foreach ( $this->enabled_metaboxes as $metabox ) {
 70             $metabox->register_metaboxes();
 71         }
 72     }
 73 
 74     /**
 75      * Saves the metabox data on post_save action
 76      *
 77      * @param $post_id the post ID
 78      */
 79     public function save_metabox_data( $post_id ) {
 80         $this->filter_metaboxes();
 81         foreach ( $this->enabled_metaboxes as $metabox ) {
 82             $metabox->save_metabox_data( $post_id );
 83         }
 84 
 85     }
 86 
 87     /**
 88      * Retrieves the values stored in the given metabox_id post meta
 89      *
 90      * @param $metabox_id the metabox ID
 91      * @param null $post_id the post ID
 92      *
 93      * @return mixed
 94      */
 95     public function get_value( $metabox_id, $post_id = null ) {
 96         foreach ( $this->metabox_infos as $metabox_info ) {
 97             if ( $metabox_info->metabox->metabox['id'] == $metabox_id ) {
 98                 return $metabox_info->metabox->get_meta( $post_id );
 99             }
100         }
101 
102     }
103 
104     /**
105      * Retrieves a list of values for the metaboxes created by the given $classname class
106      *
107      * @param $classname the name of the class
108      * @param null $post_id the post ID
109      *
110      * @return array
111      */
112     public function get_values_by_classname( $classname, $post_id = null ) {
113         $toret = array();
114         foreach ( $this->metabox_infos as $metabox_info ) {
115             if ( get_class( $metabox_info->metabox ) == $classname ) {
116                 $toret[] = $metabox_info->metabox->get_meta( $post_id );
117             }
118         }
119 
120         return $toret;
121     }
122 }
