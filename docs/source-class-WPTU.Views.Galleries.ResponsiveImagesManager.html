 67     protected function get_image($attachment_id, $size){
 68         $attachment = get_post($attachment_id);
 69         $default_attr = array(
 70                 'class'             => "attachment-$size",
 71                 'data-image-size'   =>  $size,
 72                 'data-image-id'     =>  $attachment_id,
 73                 'data-image-alt'    =>  trim(strip_tags( get_post_meta($attachment_id, '_wp_attachment_image_alt', true) )), // Use Alt field first
 74         );
 75         if ( empty($default_attr['data-image-alt']) )
 76             $default_attr['data-image-alt'] = trim(strip_tags( $attachment->post_excerpt )); // If not, Use the Caption
 77         if ( empty($default_attr['data-image-alt']) )
 78             $default_attr['data-image-alt'] = trim(strip_tags( $attachment->post_title )); // Finally, use the title
 79 
 80         return $default_attr;
 81     }

131     public function on_ajax_responsive_image(){
132         $image_id = intval($_REQUEST['id']);
133         $size = sanitize_title($_REQUEST['size']);
134         $width = intval($_REQUEST['w']);
135         
136         $path = wp_upload_dir();
137         $suitable_size =  $size.'-'.$this->get_suitable_breakpoint($width);
138         $image = image_get_intermediate_size($image_id, $suitable_size);
139         $mime_type = '';
140         
141         
142         if($image){
143             $image_path = $path['basedir'] . '/' . $image['path'];
144             $mime_type = $image['mime-type'];
145         } elseif($this->safe_mode) {
146             $image = wp_get_attachment_metadata($image_id);
147             $mime_type = get_post_mime_type($image_id);
148             $image_path = $path['basedir'] . '/' . $image['file'];
149         }
150         
151         
152         if(file_exists($image_path)){
153             header('Content-Type: ' . $mime_type);
154             header('Content-Length: ' . filesize($image_path));
155             readfile($image_path);
156         } else {
157             header("HTTP/1.0 404 Not Found");
158             wp_die(sprintf('The media size "%s" for the image with id %s does not exist', $suitable_size, $image_id));
159         }
160         
161         die();
162     }
