 13     /**
 14      * Utils to debug the code while writing it.
 15      * @author Emanuele 'Tex' Tessore
 16      * @package classes
 17      */
 18     final class DebugUtils extends Singleton {
 19 
 20         /**
 21          * @var int the level of debug: html comments, h1 and pre, h1 pre and die.
 22          */
 23         private $level;
 24 
 25         /**
 26          * @var string the title to be printed on top of the variable dump
 27          */
 28         private $title;
 29 
 30         /**
 31          * @var boolean show the file and line number in wich this feature is called
 32          */
 33         private $show_file_info;
 34 
 35         /**
 36          * @var SubstitutionTemplate the template for the debug section
 37          */
 38         public $tpl;
 39 
 40         /**
 41          * Enable or disable all debugs
 42          * @var boolean
 43          */
 44         public $status = false;
 45 
 46         /**
 47          * Wrap the var_dump into an html comment
 48          */
 49         const COMMENT = 1;
 50 
 51         /**
 52          * Print a well visible H1 and the dump is wrapped in a pre
 53          */
 54         const H1_PRE = 2;
 55 
 56         /**
 57          * Print a well visible H1, use a pre as wrapper for the variable
 58          * dump and then stop the execution of the script
 59          */
 60         const H1_PRE_DIE = 3;
 61 
 62         /**
 63          * Initializes the default settings
 64          */
 65         protected function __construct() {
 66             $this->set_level(self::COMMENT);
 67             $this->tpl = new SubstitutionTemplate(
 68                 <<<EOF
 69     <div class="debug">
 70         <h1>%title%</h1>
 71         <h2>%fileinfo%</h2>
 72         <pre>%debug%</pre>
 73     </div>
 74 EOF
 75             );
 76 
 77             if (WP_DEBUG === true)
 78                 $this->debug_deprecated();
 79 
 80 
 81         }
 82 
 83         /**
 84          * Sets the level of output.
 85          * Use DebugUtils::SOFT DebugUtils::H1_PRE or DebugUtils::H1_PRE_DIE
 86          * @param int $level the level
 87          * @return DebugUtils this for chaining
 88          */
 89         public function set_level($level) {
 90             $this->level = $level;
 91             return $this;
 92         }
 93 
 94         /**
 95          * Shows or hide info about the file and line number
 96          * @param boolean $status true if you want the infos
 97          * @return DebugUtils this for chaining
 98          */
 99         public function show_file_info($status) {
100             $this->show_file_info = $status;
101             return $this;
102         }
103 
104         /**
105          * Retrieves the current level of echo
106          * @return the current level of echo
107          */
108         public function get_level() {
109             return $this->level;
110         }
111 
112         /**
113          * Renders the file and line informations.
114          * @return string rendered info
115          */
116         public function get_file_info() {
117             $db = debug_backtrace();
118             $details = $db[2];
119             $sub = new SubstitutionTemplate();
120             return $sub->set_tpl(<<<EOF
121     <span class="file">%file%</span>:
122     <span class="line">%line%</span>
123 EOF
124             )
125                 ->set_markup('file', $details['file'])
126                 ->set_markup('line', $details['line'])
127                 ->replace_markup();
128         }
129 
130         /**
131          * Prints the dump for the given $var
132          * @param mixed $var the variable to be dumped
133          */
134         public function debug($var) {
135             if (!$this->status) return '';
136 
137             $render = $this->tpl
138                 ->set_markup('debug', var_export($var, true))
139                 ->set_markup('fileinfo', $this->get_file_info())
140                 ->set_markup('title', $this->title)
141                 ->replace_markup();
142 
143             switch ($this->level) {
144                 default:
145                 case self::COMMENT:
146                     echo '<!-- ' . $render . ' -->';
147                     break;
148 
149                 case self::H1_PRE:
150                     echo $render;
151                     break;
152 
153                 case self::H1_PRE_DIE:
154                     die($render);
155                     break;
156             }
157 
158             return $render;
159         }

166         public function set_title($title) {

174         public function dump_assets() {

183         public function debug_assets() {

187         /**
188          * Dumps the stack trace when a deprecated function is encountered
189          * @param string $function function name
190          * @param string $message deprecation message
191          * @param string $version first version with deprecated entry
192          * @todo: need improvements on readability
193          */
194         public function dump_deprecated($function, $message, $version) {
195             error_log('Deprecated Argument Detected');
196             $trace = debug_backtrace();
197             $this->debug($trace);
198             foreach ($trace as $frame) {
199                 error_log(var_export($frame, true));
200             }
201         }
202 
203         /**
204          * Enables debugging of deprecated functions
205          */
206         public function debug_deprecated() {
207             add_action('deprecated_function_run', array(&$this, 'dump_deprecated'), 10, 3);
208         }
