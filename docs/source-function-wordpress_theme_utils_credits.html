 75 if(!function_exists('get_called_class')){
 76     /**
 77      * PHP 5.2 fix for non existing get_called_class()
 78      */
 79     function get_called_class($functionName=null){
 80         $btArray = debug_backtrace();
 81         $btIndex = count($btArray) - 1;
 82         while($btIndex > -1){
 83             if(!isset($btArray[$btIndex]['file'])){
 84                 $btIndex--;
 85                 if(isset($matches[1])){
 86                     if(class_exists($matches[1])){
 87                         return $matches[1];
 88                     } else {
 89                         continue;
 90                     }
 91                 } else {
 92                     continue;
 93                 }
 94             } else {
 95                 $lines = file($btArray[$btIndex]['file']);
 96                 $callerLine = $lines[$btArray[$btIndex]['line']-1];
 97                 if(!isset($functionName)) {
 98                     preg_match('/([a-zA-Z\_]+)::/',
 99                         $callerLine,
100                         $matches);
101                 } else {
102                     preg_match('/([a-zA-Z\_]+)::'.$functionName.'/',
103                         $callerLine,
104                         $matches);
105                 }
106                 $btIndex--;
107                 if(isset($matches[1])){
108                     if(class_exists($matches[1])){
109                         return $matches[1];
110                     } else {
111                         continue;
112                     }
113                 } else {
114                     continue;
115                 }
116             }
117         }
118         return $matches[1];
119     }
120 }

122 if(!function_exists('http_build_url')){
123     /**
124      * Build a URL.
125      *
126      * The parts of the second URL will be merged into the first according to the flags argument.
127      * PECT_HTTP is usually missing on many environment.
128      * This function provides the same feature as the function included in PECT_HTTP.
129      * @see @link http://php.net/manual/en/function.http-build-url.php
130      * @param array $parsed_url the array to be merged
131      * @package core
132      * @return string the url
133      */
134     function http_build_url($parsed_url) {
135         $scheme   = isset($parsed_url['scheme']) ? $parsed_url['scheme'] . '://' : '';
136         $host     = isset($parsed_url['host']) ? $parsed_url['host'] : '';
137         $port     = isset($parsed_url['port']) ? ':' . $parsed_url['port'] : '';
138         $user     = isset($parsed_url['user']) ? $parsed_url['user'] : '';
139         $pass     = isset($parsed_url['pass']) ? ':' . $parsed_url['pass']  : '';
140         $pass     = ($user || $pass) ? "$pass@" : '';
141         $path     = isset($parsed_url['path']) ? $parsed_url['path'] : '';
142         $query    = isset($parsed_url['query']) ? '?' . $parsed_url['query'] : '';
143         $fragment = isset($parsed_url['fragment']) ? '#' . $parsed_url['fragment'] : '';
144         return "$scheme$user$pass$host$port$path$query$fragment";
145     }
146 }
