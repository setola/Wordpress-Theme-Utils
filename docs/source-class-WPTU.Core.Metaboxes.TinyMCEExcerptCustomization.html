  8 /**
  9  * This class removes the default excerpt metabox
 10  * and adds a new box with the wysiwyg editor capability
 11  * @author etessore
 12  * @version 1.0.0
 13  * @package classes
 14  */
 15 class TinyMCEExcerptCustomization{
 16     const textdomain = '';
 17     const custom_exceprt_slug = '_custom-excerpt';
 18     
 19     /**
 20      * @var array a list of context where wysiwyg editor is required
 21      */
 22     var $contexts;
 23 
 24     /**
 25      * Set the feature up
 26      * @param array $contexts a list of context where you want the wysiwyg editor in the excerpt field. Defatul array('post','page')
 27      */
 28     function __construct($contexts=array('post', 'page')){
 29 
 30         $this->contexts = $contexts;
 31 
 32         add_action('admin_menu', array($this, 'remove_excerpt_metabox'));
 33         add_action('add_meta_boxes', array($this, 'add_tinymce_to_excerpt_metabox'));
 34         add_filter('wp_trim_excerpt',  array($this, 'custom_trim_excerpt'), 10, 2);
 35         add_action('save_post', array($this, 'save_box'));
 36     }
 37 
 38     /**
 39      * Removes the default editor for the excerpt
 40      */
 41     function remove_excerpt_metabox(){
 42         foreach($this->contexts as $context)
 43             remove_meta_box('postexcerpt', $context, 'normal');
 44     }
 45 
 46     /**
 47      * Adds a new excerpt editor with the wysiwyg editor
 48      */
 49     function add_tinymce_to_excerpt_metabox(){
 50         foreach($this->contexts as $context)
 51             add_meta_box(
 52                     'tinymce-excerpt',
 53                     __('Excerpt', self::textdomain),
 54                     array($this, 'tinymce_excerpt_box'),
 55                     $context,
 56                     'normal',
 57                     'high'
 58             );
 59     }
 60 
 61     /**
 62      * Manages the excerpt escaping process
 63      * @param string $text the default filtered version
 64      * @param string $raw_excerpt the raw version
 65      */
 66     function custom_trim_excerpt($text, $raw_excerpt) {
 67         global $post;
 68         $custom_excerpt = get_post_meta($post->ID, self::custom_exceprt_slug, true);
 69         if(empty($custom_excerpt)) return $text;
 70         return $custom_excerpt;
 71     }
 72 
 73     /**
 74      * Prints the markup for the tinymce excerpt box
 75      * @param object $post the post object
 76      */
 77     function tinymce_excerpt_box($post){
 78         $content = get_post_meta($post->ID, self::custom_exceprt_slug, true);
 79         if(empty($content)) $content = '';
 80         wp_editor(
 81                 $content,
 82                 self::custom_exceprt_slug,
 83                 array(
 84                         'wpautop'       =>  true,
 85                         'media_buttons' =>  false,
 86                         'textarea_rows' =>  10,
 87                         'textarea_name' =>  self::custom_exceprt_slug
 88                 )
 89         );
 90     }
 91 
 92     /**
 93      * Called when the post is beeing saved
 94      * @param int $post_id the post id
 95      */
 96     function save_box($post_id){
 97         update_post_meta($post_id, self::custom_exceprt_slug, $_POST[self::custom_exceprt_slug]);
 98     }
 99 }
