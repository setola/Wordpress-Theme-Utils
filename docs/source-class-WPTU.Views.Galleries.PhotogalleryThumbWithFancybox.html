 12  * Manages images and markup for a photogallery where 
 13  * clicking on a little thumb opens the big image in fancybox

 22 /*
 23  * Changelog
 24  *  1.0.1
 25  *   removed dependency from FeatureWithAssets, use ThemeHelpers::load_js() instead
 26  */
 27 class PhotogalleryThumbWithFancybox extends GalleryHelper{
 28     
 29     /**
 30      * @var array stores the sizes of the images
 31      */
 32     public $sizes;
 33     
 34     /**
 35      * The number of images per row
 36      * @var int
 37      */
 38     public $images_per_row;
 39     
 40     /**
 41      * The image container class
 42      * @var string
 43      */
 44     public $single_image_container_class;
 45     
 46     /**
 47      * Stores the html template
 48      * @var SubstitutionTemplate
 49      */
 50     public $subs;
 51 
 52     /**
 53      * Initializes the photogallery
 54      */
 55     public function __construct(){
 56         $this->subs = new SubstitutionTemplate();
 57         $this->subs
 58             ->set_tpl(<<< EOF
 59     <div id="%gallery-id%" class="gallery">
 60         <div class="big-image-container">
 61             %images%
 62         </div>
 63     </div>
 64 EOF
 65             )
 66             ->set_markup('loading', '<div class="loading">'.__('Loading...', 'wtu_framework').'</div>');
 67         $this
 68             ->set_wp_media_dimension('photogallery')
 69             ->set_images_per_row(4)
 70             ->set_single_image_container_class('grid_4');
 71     }
 72     
 73     /**
 74      * Sets the number of images per row
 75      * @param int $images_per_row a number
 76      * @return PhotogalleryThumbWithFancybox $this for chainability
 77      */
 78     public function set_images_per_row($images_per_row){
 79         $this->images_per_row = $images_per_row;
 80         return $this;
 81     }
 82     
 83     /**
 84      * Set the class for single image container
 85      * @param string $class the class
 86      * @return PhotogalleryThumbWithFancybox $this for chainability
 87      */
 88     public function set_single_image_container_class($class){
 89         $this->single_image_container_class = $class;
 90         return $this;
 91     }
 92     
 93     /**
 94      * Sets the images sizes: array('w'=>xxx, 'h'=>yyy)
 95      * @param array $sizes
 96      * @return PhotogalleryThumbWithFancybox $this for chainability
 97      */
 98     public function set_sizes($sizes){
 99         $this->sizes = $sizes;
100         return $this;
101     }
102     
103     /**
104      * (non-PHPdoc)
105      * @see GalleryHelper::get_markup()
106      */
107     public function get_markup(){
108         $markup_images = '';
109         if(count($this->images)>0){
110             $images_per_line = $this->images_per_row;
111             foreach ($this->images as $k => $image){
112                 $classes = array($this->single_image_container_class, 'image_'.$k);
113                 
114                 if($images_per_line == 0){
115                     $images_per_line = $this->images_per_row;
116                 }
117                 
118                 if($images_per_line == $this->images_per_row){
119                     $classes[] = 'alpha';
120                 }
121                 if($images_per_line == 1){
122                     $classes[] = 'omega';
123                 }
124                 $images_per_line--;
125                 
126                 $big_img_src = wp_get_attachment_image_src($this->get_image_id($k), 'full');
127                 $big_img_src = $big_img_src[0];
128                 
129                 
130                 $markup_images .= 
131                     '<div class="'.implode(' ', $classes).'">'.
132                     HtmlHelper::anchor(
133                         $big_img_src,
134                         HtmlHelper::image(
135                             $this->get_image_src($k),
136                             array(
137                                 'alt'           =>  $this->get_image_alt($k)    
138                             )
139                         ),
140                         array(
141                             'rel'               =>  'group', 
142                             'class'             =>  'fancy', 
143                             'title'             =>  $this->get_image_title($k),
144                             'data-description'  =>  $this->get_image_description($k),
145                             'data-caption'      =>  $this->get_image_caption($k)
146                         )
147                     )
148                     .'</div>';
149             }
150         }
151     
152         if(!$this->unid){
153             $this->calculate_unique();
154         }
155         
156         $this->subs->set_markup('gallery-id', $this->unid);
157         $this->subs->set_markup('images', $markup_images);
158         
159         ThemeHelpers::load_js('jquery-fancybox');
160         ThemeHelpers::load_css('jquery-fancybox');
161         
162         return $this->subs->replace_markup();
163     }
164 }
