 43     /**
 44      * Initializes the blog with starting data.
 45      * 
 46      * Overload this in a child ob ject if you need customization on starting data
 47      * 
 48      * @param int $blog_id the blog id
 49      */
 50     public function init($blog_id){
 51         //make sure we are in the right blog!
 52         $this->blog_id = $blog_id;
 53         switch_to_blog($this->blog_id);
 54         
 55         $this->delete_default_posts();
 56         $this->add_theme();
 57         
 58         // create the main menu if not exists
 59         $top_menu_id = $this->create_menu('Main Menu');
 60         
 61         // and add some pages to it
 62         $this->add_page_to_menu($top_menu_id, $this->add_page('Home'));
 63         $this->add_page_to_menu($top_menu_id, $this->add_page('Rooms'));
 64         $this->add_page_to_menu($top_menu_id, $this->add_page('Location'));
 65     }

 67     /**
 68      * Creates the menu with given name
 69      * 
 70      * If a menu with given name exists this method will return its id.
 71      * If $location parameter is passed this method will assign the menu with given $name to such location
 72      * 
 73      * @param string $name the menu name
 74      * @param string $location
 75      * @return Ambigous <mixed, boolean, unknown, multitype:, NULL, WP_Error, object, array|object, string, number, error, multitype:mixed unknown , multitype:number mixed , multitype:number Ambigous <string, NULL> >
 76      */
 77     public function create_menu($name, $location=null){
 78         $id = wp_get_nav_menu_object($name)
 79             ? wp_get_nav_menu_object($name)
 80             : wp_create_nav_menu($name);
 81         
 82         if(!(is_null($location) || has_nav_menu($location))){
 83             $locations = get_theme_mod('nav_menu_locations');
 84             $locations[$location] = $id;
 85             set_theme_mod('nav_menu_locations', $locations);
 86         }
 87         
 88         return is_object($id) ? intval($id->term_id) : $id;
 89     }

113     /**
114      * Inserts a new page into WP
115      * @param string|array $parms page details according to {@link http://codex.wordpress.org/Function_Reference/wp_insert_post wp_insert_post()}
116      * @return int the post id
117      */
118     public function add_page($parms=null){
119         // make sure we are in the right blog!
120         $this->select_blog();
121         
122         // pass a string as parameter and it will be the post title
123         if(is_string($parms)){
124             $parms = array('post_title' => $parms);
125             $parms['post_name'] = sanitize_title($parms['post_title']);
126         }
127         
128         // che for template existence
129         if(!empty($parms['page_template'])){
130             $parms['page_template'] = locate_template($parms['page_template']);
131         }
132 
133         $defaults = array(
134             'post_parent'   =>  0,
135             'post_status'   =>  'publish',
136             'post_title'    =>  'New Automatic Page',
137             'post_name'     =>  'new-automatic-page',
138             'page_template' =>  '',
139             'post_type'     =>  'page'
140         );
141         $args = wp_parse_args((array)$parms, $defaults);
142         
143         $args['post_id'] = wp_insert_post($args);
144 
145         if(!$args['post_id']) wp_die(sprintf(__('Unable to insert page "%s"'), $args['post_name']));
146         
147         $this->pages[$args['post_id']] = $args;
148         return $args['post_id'];
149     }

160     /**
161      * Adds an automatically generated page to the given menu
162      * @param int $index the index of $this->pages array
163      * @param int $menu_id the menu id
164      * @return NewBlogInitializator $this for chainability

166     public function add_menu_entry($index, $menu_id){
167         if(!isset($this->pages[$index]))
168             wp_die(sprintf(__('Unable to insert page number "%s" into menu %s: <br />This page wasn\'t created by automatic content filler subsystem!'), $index, $menu_id));
169         
170         $parms = array(
171             'menu-item-title'       =>  $this->pages[$index]['post_title'],
172             'menu-item-object'      =>  'page',
173             'menu-item-object-id'   =>  $this->pages[$index]['post_id'],
174             'menu-item-type'        =>  'post_type',
175             'menu-item-classes'     =>  $this->pages[$index]['post_name'],
176             'menu-item-status'      =>  'publish'
177         );
178         
179         return wp_update_nav_menu_item($menu_id, 0, $parms);
180     }

208     /**
209      * Creates the basic structure for a child theme.
210      *
211      * This method tries to make a new theme directory
212      * under the wp-contents/themes.
213      * Then tries to create style.css in the newly
214      * theme folder and fill it with the default header comment
215      *
216      * An example of parameters:
217      * <code>
218      * $defaults = array(
219      *      'name'          =>  'Wordpress Theme Utils Child Theme',
220      *      'folder'        =>  'wtu-child-theme',
221      *      'child_of'      =>  'Wordpress-Theme-Utils',
222      *      'description'   =>  'This is a Child Theme of Wordpress Theme Utils',
223      *      'theme_uri'     =>  'http://www.emanueletessore.com/',
224      *      'author'        =>  'Emanuele Tessore',
225      *      'author_uri'    =>  'http://www.emanueletessore.com/',
226      *      'version'       =>  '1.0.0'
227      * );
228      * </code>
229      *
230      * Uses {@link http://codex.wordpress.org/Function_Reference/wp_die wp_die()} if something goes wrong
231      *
232      * @param array $parms parameters for the new theme
233      * @return NewBlogInitializator $this for chainability
234      */
235     public function add_theme($parms=null){
236         $defaults = array(
237             'name'          =>  'Wordpress Theme Utils Child Theme',
238             'folder'        =>  'wtu-child-theme',
239             'child_of'      =>  'Wordpress-Theme-Utils',
240             'description'   =>  'This is a Child Theme of Wordpress Theme Utils',
241             'theme_uri'     =>  'http://www.emanueletessore.com/',
242             'author'        =>  'Emanuele Tessore',
243             'author_uri'    =>  'http://www.emanueletessore.com/',
244             'version'       =>  '1.0.0'
245         );
246         $args = wp_parse_args((array)$parms, $defaults);
247 
248         // remove spaces and html\php tags
249         $new_dir_path = get_theme_root().'/'.sanitize_title($args['folder']);
250         $new_style_path = $new_dir_path.'/style.css';
251 
252         // this is the default header for the style.css
253         $new_style_content = <<< EOF
254 /*
255 THEME NAME:      {$args['name']}
256 THEME URI:       {$args['theme_uri']}
257 VERSION:         {$args['version']}
258 AUTHOR:          {$args['author']}
259 AUTHOR URI:      {$args['author_uri']}
260 DESCRIPTION:     {$args['description']}
261 TEMPLATE:        {$args['child_of']}
262 */
263 
264 @import url("../{$args['child_of']}/style.css");
265 EOF;
266 
267         // check if the directory we want to create exists
268         if(is_dir($new_dir_path)){
269             // if so we cannot proceed or we will overwrite another theme
270             wp_die(sprintf(__('Unable to create the new theme:<br />Directory <code>%s</code> exists'), $new_dir_path));
271         }
272 
273         // try to make the new directory
274         if(mkdir($new_dir_path)===false){
275             // we had an error while creating the directory, we cannot proceed
276             wp_die(sprintf(__('Unable to create the new theme:<br />Error while creating the directory <code>%s</code>'), $new_dir_path));
277         }
278 
279         // test if the style.css exists
280         if(file_exists($new_style_path)){
281             // if so we cannot proceed or we will overwrite the stylesheet
282             wp_die(sprintf(__('Unable to create the new theme:<br />Error <code>%s</code> already exists'), $new_style_path));
283         }
284 
285         // try to write the content into style.css
286         if(file_put_contents($new_style_path, $new_style_content)===false){
287             // if the content insertion failed we cannot proceed
288             wp_die(sprintf(__('Unable to create the new theme:Error while writing <code>%s</code>'), $new_style_path));
289         }
290 
291         // if we reached this point everything is gone the right way
292         $this->theme = $args;
293         return $this;
294     }

296     /**
297      * Enables the new theme for the current blog
298      * @return NewBlogInitializator $this for chainability
299      */
300     public function enable_theme(){
301         $this->select_blog();
302         
303         if(is_multisite()){
304             // Enable Child Theme for this site only
305             $allowed_themes = get_option('allowedthemes');
306             
307             if ( !$allowed_themes ){
308                 $allowed_themes = array( $folder => true );
309             } else {
310                 $allowed_themes[$folder] = true;
311             }
312             
313             update_option('allowedthemes', $allowed_themes);
314         }
315         
316         // Activate child theme on this site
317         switch_theme($this->theme['child_of'], $this->theme['folder']);
318         
319         return $this;
320     }

323      * Deletes all pages
324      * 
325      * WARNING: it does NOT ask if you're sure!!!
326      * 
327      * @return NewBlogInitializator $this for chainability
328      */
329     public function delete_all_pages(){
330         $this->select_blog();
331         
332         $options = array(
333                 'numberposts'     => 50,
334                 'offset'          => 0,
335                 'orderby'         => 'post_date',
336                 'order'           => 'DESC',
337                 'post_type'       => 'page',
338         );
339         
340         $posts = get_posts( $options );
341         
342         $offset = 0;
343         while( count( $posts ) > 0 ) {
344             if( $offset == 10 ) {
345                 break;
346             }
347             $offset++;
348             
349             foreach( $posts as $post ) {
350                 wp_delete_post( $post->ID, true );
351             }   
352             $posts = get_posts( $options );
353         }
354         
355         return $this;
356     }
