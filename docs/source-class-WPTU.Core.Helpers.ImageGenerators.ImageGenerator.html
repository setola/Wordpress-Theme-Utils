 45     /**
 46      * Initializes the object by checking 
 47      * if the image exists or has to be generated
 48      */
 49     public function init(){
 50         $path = wp_upload_dir();
 51         
 52         $subdir = '/placeholders/';
 53         
 54         $this->cache['path']    =   $path['basedir'] . $subdir;
 55         $this->cache['name']    =   md5(serialize($this->settings)).'.png';
 56         $this->cache['url']     =   $path['baseurl'] . $subdir . $this->cache['name'];
 57         
 58         if(!is_dir($this->cache['path'])){
 59             mkdir($this->cache['path']);
 60         }
 61         
 62         if(!file_exists($this->cache['path'].$this->cache['name']) && empty($this->image)) {
 63             $this->generate_image()->save_image();
 64         }
 65         
 66         return $this;
 67     }

104     /**
105      * Generates an image
106      * @throw Exception if GD Libs are not installed on your host
107      * @return ImageGenerator $this for chainability
108      */
109     private function generate_image(){
110         if(!function_exists('imagecreate')){
111             throw new \Exception('GD Libs are missing. Please install them on your host server.');
112         }
113         $this->image = imagecreate($this->get_image_width(), $this->get_image_height());
114         
115         $bg_color = self::get_rgb($this->settings['bg_color']);
116         $text_color = self::get_rgb($this->settings['text_color']);
117         
118                 
119         imagefill(
120             $this->image, 
121             0, 
122             0, 
123             imagecolorallocate(
124                 $this->image, 
125                 $bg_color['red'], 
126                 $bg_color['green'], 
127                 $bg_color['blue']
128             )
129         );
130         
131         $border = imagecolorallocate(
132             $this->image, 
133             $text_color['red'], 
134             $text_color['green'], 
135             $text_color['blue']
136         );
137         imagerectangle(
138             $this->image, 
139             0, 0, 
140             $this->get_image_width() - 1, 
141             $this->get_image_height() - 1, 
142             $border
143         );
144         
145         $fontsize = $this->settings['font_size'];
146         if(empty($fontsize) || !is_numeric($fontsize)) $fontsize = 
147             ($this->settings['width'] > $this->settings['height'])
148             ? ($this->get_image_height() / 10) 
149             : ($this->get_image_width() / 10);
150         
151         $image_text = (isset($this->settings['image_text']))
152             ? $this->settings['image_text']
153             : $this->get_image_width().' X '.$this->get_image_height();
154         
155         $text_position_x = $this->settings['text_position_x'];
156         if(empty($text_position_x) || !is_numeric($text_position_x))
157             $text_position_x = ($this->get_image_width()/2) - ($fontsize * 2.75);
158         
159         $text_position_y = $this->settings['text_position_y'];
160         if(empty($text_position_y) || !is_numeric($text_position_y))
161             $text_position_y = ($this->get_image_height()/2) + ($fontsize* 0.2);
162         
163         imagettftext(
164             $this->image,
165             $fontsize, 
166             0,
167             $text_position_x,
168             $text_position_y,
169             imagecolorallocate($this->image, $text_color['red'], $text_color['green'], $text_color['blue']), 
170             WORDPRESS_THEME_UTILS_PATH . '/Fonts/DS-Digital.ttf',
171             $image_text
172         );
173         
174         return $this;
175     }

223     /**
224      * Hooks into WordPress to serve a generated image when no one is available
225      */
226     public function hook(){
227         get_the_post_thumbnail();
228         add_filter( "get_post_metadata", array($this, 'get_post_metadata_callback'), 10, 4);
229         add_action( "post_thumbnail_html", array($this, 'post_thumbnail_html_callback'), 10, 5);
230     }

232     /**
233      * Callback for 'get_post_metadata' hook.
234      * Returns -1 if post image is not set, else the id of the image
235      * @param int $value the ID of the post thumbnail
236      * @param int $object_id ID of the object metadata is for
237      * @param string $meta_key Optional.  Metadata key.  
238      *      If not specified, retrieve all metadata for the specified object.
239      * @param bool $single Optional, default is false.  
240      *      If true, return only the first value of the specified meta_key.  
241      *      This parameter has no effect if meta_key is not specified.
242      */
243     public function get_post_metadata_callback($value, $object_id, $meta_key, $single){
244         if($meta_key=='_thumbnail_id' && is_null($value)){
245             return -1;
246         }
247         
248         return $object_id;
249     }

251     /**
252      * Callback for 'post_thumbnail_html' hook.
253      * Changes the html markup if the post thumbnail is not set
254      * @param string $html
255      * @param unknown_type $post_id
256      * @param unknown_type $post_thumbnail_id
257      * @param unknown_type $size
258      * @param unknown_type $attr
259      */
260     public function post_thumbnail_html_callback($html, $post_id, $post_thumbnail_id, $size, $attr){
261         if($post_thumbnail_id==-1){
262             $sizes = self::get_all_image_sizes();
263             if(empty($size) || !in_array($size, array_keys($sizes))){
264                 $size = 'medium';
265             }
266             return HtmlHelper::image(
267                 admin_url('admin-ajax.php')
268                 .'?'.build_query(array(
269                     'action'    =>  'placeholder',
270                     'w'         =>  $sizes[$size]['width'],
271                     'h'         =>  $sizes[$size]['height']
272                 ),
273                 array(
274                     'alt'       =>  __('Placeholder post thumbnail'),
275                     'width'     =>  $sizes[$size]['width'],
276                     'height'    =>  $sizes[$size]['height']     
277                 ))
278             );
279         }
280         return $html;
281     }
