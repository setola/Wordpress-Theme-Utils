 90     /**
 91      * Saves the metabox data while saving the page
 92      */
 93     public static function save_metabox_data($post_id){
 94         if(!isset($post_id)) return;
 95         if(!isset($_POST['post_type'])) return;
 96         // First we need to check if the current user is authorised to do this action.
 97         if ( 'page' == $_POST['post_type'] ) {
 98             if ( ! current_user_can( 'edit_page', $post_id ) )
 99                 return;
100         } else {
101             if ( ! current_user_can( 'edit_post', $post_id ) )
102                 return;
103         }
104 
105         // Secondly we need to check if the user intended to change this value.
106         if ( 
107             !isset($_POST[self::META_KEY_NAME.'_nonce']) 
108             || !wp_verify_nonce($_POST[self::META_KEY_NAME.'_nonce'], __FILE__) 
109         ) {
110             return;
111         }
112         
113         update_post_meta(
114             $_POST['post_ID'], 
115             self::META_KEY_NAME, 
116             sanitize_text_field($_POST[self::META_KEY_NAME])
117         );
118     }

137     public static function get_hotels_ids(){
138         global $wpdb;
139         $ids = $wpdb->get_col('SELECT post_id FROM  `'.$wpdb->postmeta.'` WHERE  `meta_key` = \''.self::META_KEY_NAME.'\' AND `meta_value` = \'on\'');
140         
141         $toret = array();
142         
143         if(function_exists('icl_object_id')){
144             foreach ($ids as $k => $id) {
145                 $xlat = icl_object_id($id, get_post_type($id), false);
146                 if(!is_null($xlat)) $toret[$xlat] = $xlat;
147             }
148             return array_keys($toret);
149         }
150         
151         return $ids;
152     }
