 23 class EscapeRoute{
 24     /**
 25      * @var array the list of link to be printed in case of 404
 26      */
 27     public $links = array();
 28     
 29     /**
 30      * @var string the substitution template
 31      */
 32     public $tpl;
 33     
 34     /**
 35      * @var SubstitutionTemplate the substitution engine
 36      */
 37     public $templates;
 38     
 39     /**
 40      * @var string the list separator
 41      */
 42     public $list_separator = '';
 43     
 44     /**
 45      * Initializes the object to the default values
 46      */
 47     public function __construct(){
 48         $tpl = <<<EOF
 49         <h1>%title%</h1>
 50         <h2>%subtitle%</h2>
 51         <p>%desc%</p>
 52         %list%
 53 EOF;
 54         $this->templates = new SubstitutionTemplate();
 55         $this->templates
 56             ->set_tpl($tpl)
 57             ->set_markup('title', __('The page was not found', 'theme'))
 58             ->set_markup('subtitle', __('I\'m sorry but the page you were looking for is not available', 'theme'))
 59             ->set_markup('desc', __('Maybe one of these links will be usefull', 'theme'));
 60         
 61         $this
 62             ->set_template($tpl)
 63             ->set_separator(' - ')
 64             ->add_link('javascript:history.back();', __('Back', 'theme'))
 65             ->add_link(home_url(), __('Homepage', 'theme'))
 66             ->add_link('javascript.void();', __('Book Now', 'theme'), array('class'=>'book-action2'));
 67     }
 68     
 69     /**
 70      * Adds a link
 71      * @param string $href the url to be linked
 72      * @param string $label the inner html for the <a> tag
 73      * @param string|array $params additional parameters for the <a>
 74      * @return EscapeRoute $this for chainability
 75      */
 76     public function add_link($href, $label, $params=''){
 77         $tmp            = new \stdClass();
 78         $tmp->href      = $href;
 79         $tmp->label     = $label;
 80         $tmp->params    = $params;
 81         $this->links[$href] = $tmp;
 82         return $this;
 83     }
 84     
 85     /**
 86      * Set the html template
 87      * @param string $tpl the template
 88      * @return EscapeRoute $this for chainability
 89      */
 90     public function set_template($tpl){
 91         $this->tpl = $tpl;
 92         return $this;
 93     }
 94     
 95     /**
 96      * Set the separator for the list of anchors
 97      * @param string $separator html between every <a>
 98      */
 99     public function set_separator($separator){
100         $this->list_separator = $separator;
101         return $this;
102     }
103     
104     /**
105      * Renders the list of anchors
106      * @return string the rendered the list of anchors
107      */
108     protected function render_list(){
109         $toret = '';
110         $first = true;
111         foreach($this->links as $link){
112             if(!$first){
113                 $toret .= $this->list_separator;
114             }
115             $toret .= HtmlHelper::anchor($link->href, $link->label, $link->params);
116             $first = false;
117         }
118         return $toret;
119     }
120     
121     /**
122      * Retrieves the markup
123      * @returns string the markup
124      */
125     public function get_markup(){
126         return $this->templates
127             ->set_markup('list', $this->render_list())
128             ->replace_markup();
129     }
