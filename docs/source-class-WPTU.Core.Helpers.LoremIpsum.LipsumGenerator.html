232     /**
233      * Retrieves an array of random words
234      * @return array a list of random words
235      * @param $count int the number of words to ber retrieved
236      */
237     private function get_words($count){
238         $toret = array();
239         $words = $this->dictionary->get_all_words();
240 
241         while(count($toret)<$count){
242             //$word  = $words[array_rand($words)];
243             $word  = $words[rand(0, $this->dictionary->number_of_words()-1)];
244             
245             // do not repeat the same word in a 5 words group
246             if(!in_array($word, array_slice($toret, $this->min_repeat_count*-1))){
247                 $toret[] = $word;
248             }
249         }
250 
251         return $toret;
252     }

254     /**
255      * Returns a number on a gaussian distribution based
256      * on the average word length of an english sentence.
257      * Statistics Source:
258      *  http://hearle.nahoo.net/Academic/Maths/Sentence.html
259      *  Average: 24.46
260      *  Standard Deviation: 5.08
261      */
262     private function gaussian_sentence(){
263         $avg        =   (float) 24.460;
264         $std_dev    =   (float) 5.080;
265         return (int) round($this->gaussian_math->gauss_ms($avg, $std_dev));
266     }

268     /**
269      * Builds a single paragraph
270      */
271     private function get_paragraph(){
272         $count = $this->words_per_paragraph;
273         $words = $this->get_words($count);
274 
275         $delta = $count;
276         $curr = 0;
277         $sentences = array();
278         while ($delta > 0) {
279             $senSize = $this->gaussian_sentence();
280 
281             if (($delta - $senSize) < 4) {
282                 $senSize = $delta;
283             }
284 
285             $delta -= $senSize;
286 
287             $sentence = array();
288             for ($i = $curr; $i < ($curr + $senSize); $i++) {
289                 $sentence[] = $words[$i];
290             }
291             $sentence[0] = ucfirst($sentence[0]);
292 
293             $this->punctuate($sentence);
294             $curr = $curr + $senSize;
295             $sentences[] = $sentence;
296         }
297 
298         return $sentences;
299     }

469     /**
470      * Generates and renders in rich html format
471      * @see LipsumGenerator::FORMAT_RICH_HTML
472      * @return LipsumGenerator $this for chainability
473      */
474     private function render_rich_html(){
475         $this->render = '';
476         $is_first_p = true;
477         $this->stats = array();
478         // useful to avoid the same tag to be repeated without plain text between
479         $last_closed_tag = false;
480         
481         if(empty($this->rich_html_config)) $this->set_rich_html_config();
482         
483         for($i=0; $i<$this->number_of_paragraphs; $i++){
484             $paragraph = $this->get_paragraph();
485             $rendered_p = '';
486             
487             if($is_first_p){
488                 $this->force_beginning($paragraph);
489             } else {
490                 $rendered_p .= "\n";
491             }
492             
493             $rendered_p .= HtmlHelper::open_tag('p'); //'<p>';
494             foreach($paragraph as $sentence){
495                 // stores the currently opened html tag
496                 $current_tag = '';
497                 // stores how many words are left before the current tag have to be closed
498                 $words_until_close_tag = 0;
499                 foreach($sentence as $index => $word){
500                     if($words_until_close_tag < 0){
501                         $words_until_close_tag = 0;
502                     }
503                     
504                     if(empty($current_tag) && empty($last_closed_tag)){
505                         $tag_to_add = '';
506                         foreach($this->rich_html_config as $tag => $tag_config){
507                             $rnd = rand(0, 99);
508                             if($rnd < $tag_config['percent']){
509                                 //v(array($rnd, $tag, $index));
510                                 if(
511                                         isset($this->rich_html_config[$tag_to_add]['percent']) 
512                                         && $this->rich_html_config[$tag_to_add]['percent'] < $tag_config['percent']
513                                 ){
514                                     $tag_to_add = $tag;
515                                     $this->stats[$tag]++;
516                                 }
517                             }
518                         }
519                     }
520                     
521                     if(!empty($tag_to_add)){
522                         // insert the current tag!
523                         $current_tag = $tag_to_add;
524                         //$rendered_p .= $this->open_tag($tag_to_add);
525                         $rendered_p .= HtmlHelper::open_tag($tag_to_add, $this->rich_html_config[$tag_to_add]['params']);
526                         $words_until_close_tag = rand(1, $this->rich_html_config[$tag_to_add]['max_words']);
527                         $tag_to_add = '';
528                     }
529                     
530                     
531                     $rendered_p .= $word;
532                     
533                     $last_closed_tag = '';
534                     
535                     if(!empty($current_tag)){
536                         if($words_until_close_tag == 0){
537                             $rendered_p .= HtmlHelper::close_tag($current_tag);
538                             $last_closed_tag = $current_tag;
539                             $current_tag = '';
540                         } else {
541                             $words_until_close_tag--;
542                         }
543                     }
544                     
545                     $rendered_p .= ' ';
546                     
547                     
548                 }
549                 // close the last tag if it's still open at the paragraph end
550                 $rendered_p .= HtmlHelper::close_tag($current_tag);
551             }
552             $rendered_p .= HtmlHelper::close_tag('p'); //"</p>";
553             
554             $this->render .= $rendered_p;
555             $is_first_p = false;
556         }
557         return $this;
558     }

575     /**
576      * Inserts commas and periods in the given array of words.
577      * @param array $sentence the list of sentence
578      */
579     private function punctuate(& $sentence){
580         $count = count($sentence);
581         $sentence[$count - 1] .= '.';
582     
583         if ($count < 4) {
584             return $sentence;
585         }
586     
587         $commas = $this->numberOfCommas($count);
588     
589         for ($i = 1; $i <= $commas; $i++) {
590             $index = (int) round($i * $count / ($commas + 1));
591     
592             if ($index < ($count - 1) && $index > 0) {
593                 $sentence[$index] .= ',';
594             }
595         }
596     }

598     /**
599      * Determines the number of commas for a
600      * sentence of the given length. Average and
601      * standard deviation are determined superficially
602      * @param $len int a medium amount of words between two commas
603      */
604     private function numberOfCommas($len){
605         $avg    = (float) log($len, 6);
606         $stdDev = (float) $avg / 6.000;
607     
608         return (int) round($this->gaussian_math->gauss_ms($avg, $stdDev));
609     }
