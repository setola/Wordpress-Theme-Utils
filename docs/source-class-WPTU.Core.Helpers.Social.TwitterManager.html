 48      * Some transients for cache

 51     public static $transients = array(

 59      * List of WP AJAX actions used by this feature
 60      * @var array
 61      */
 62     public static $actions = array(
 63         'update_tweets' =>  array(
 64             'action'        =>  'update_tweets',
 65             'callback'      =>  array(__CLASS__, 'on_ajax_update_tweets'),
 66         ),
 67         'show_tweets'   =>  array(
 68             'action'        =>  'show_tweets',
 69             'callback'      =>  array(__CLASS__, 'on_ajax_show_tweets'),
 70         ),
 71     );
 72     
 73     /**
 74      * Enables the connection with Twitter
 75      * 
 76      * You need to create a new app and authorize it from your twitter account
 77      * 
 78      * @param string $customer_key Consumer key
 79      * @param string $customer_secret Consumer secret
 80      * @param string $token Access token
 81      * @param string $token_secret Access token secret
 82      */
 83     public static function enable($customer_key, $customer_secret, $token, $token_secret){
 84         self::$oauth_key            =   $customer_key;
 85         self::$oauth_secret         =   $customer_secret;
 86         self::$oauth_token          =   $token;
 87         self::$oauth_token_secret   =   $token_secret;
 88         
 89         foreach(self::$actions as $action){
 90             add_action('wp_ajax_'.$action['action'], $action['callback']);
 91             add_action('wp_ajax_nopriv_'.$action['action'], $action['callback']);
 92         }
 93         
 94         require_once WORDPRESS_THEME_UTILS_LIBRARIES_ABSOLUTE_PATH . 'twitteroauth/twitteroauth/twitteroauth.php';
 95         
 96         self::$connection = new TwitterOAuth(
 97             self::$oauth_key, 
 98             self::$oauth_secret, 
 99             self::$oauth_token, 
100             self::$oauth_token_secret
101         );
102     }
103     
104     /**
105      * Completely disables the connection with Twitter
106      */
107     public static function disable(){
108         foreach(self::$actions as $action){
109             remove_action('wp_ajax_'.$action['action'], $action['callback']);
110             remove_action('wp_ajax_nopriv_'.$action['action'], $action['callback']);
111         }
112     }
113     
114     /**
115      * Updates the list of tweets
116      */
117     public static function on_ajax_update_tweets(){
118         $result = self::$connection->get('statuses/user_timeline');
119         if(isset($result->errors)){
120             wp_mail(
121                 get_option('admin_email'), 
122                 __('Error on Twitter Manager', 'wtu_framework'), 
123                 '<pre>'.$result.'</pre>'
124             );
125         }
126         
127         set_transient(self::$transients['tweets'], $result);
128 
129         header('Content-type: application/json');
130         die(json_encode($result));
131     }
132     
133     /**
134      * Shows the cached tweets
135      */
136     public function on_ajax_show_tweets(){
137         header('Content-type: application/json');
138         die(json_encode(get_transient(self::$transients['tweets'])));
139     }
